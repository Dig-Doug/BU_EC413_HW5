int genRandom() {

	TA0CTL |= TACLR;            // reset clock
	TA0CTL = TASSEL_2+ID_0+MC_1;// clock source = SMCLK
	                            // clock divider=1
	                            // UP mode
	                            // timer A interrupt off
	// compare mode, output mode 0, no interrupt enabled
	TA0CCTL0=0;
	TA0CCR0 = -1;
	
	volatile int i = 0;
	while (i < 1000) {
		i++;
	}

	int result = TAR0;
	TA0CTL |= TACLR;
	
	return result;
}
